                function crearGaleria() {
                    //-- Seleccionar lista donde se insertar√° la galer√≠a 
                    const galeria = document.querySelector('.galeria-imagenes');

                    
                    for( let i=1; i<=18; i++ ) {
                        //-- Crear la etiqueta <img> y asignarle su respectivo src: imagen thumbnail
                        const imagen = document.createElement('IMG');
                        imagen.src = `img/thumbnail/barcelona-thumb-${i}.jpg`;
                        //console.log(imagen);

                        //-- Crear la etiqueta <li> y apilar dentro imagen
                        const lista = document.createElement('LI');
                        lista.appendChild(imagen);

                        galeria.appendChild(lista);

                        //-- A√±adir la funci√≥n de mostrar imagen
                        imagen.onclick = mostrarImagen;
                        imagen.dataset.imagenId = i;
                    }
                }

‚ñ™Ô∏è Funci√≥n Mostrar Galer√≠a

              
                function mostrarImagen(e) {
                    //console.log("Diste click en la imagen");
                    //console.log(e.target.dataset.imagenId);
                    
                    //-- Recuperar el id de la imagen a la que se di√≥ click
                    const id = parseInt(e.target.dataset.imagenId);
                    //console.log(id);

                    //-- Crear la etiqueta <img> y asignarle su respectivo src: imagen grande
                    const imagen = document.createElement('IMG');
                    imagen.src = `img/complete/barcelona-${id}.jpg`;
                    //console.log(imagen);

                    //-- Crear el <div> Overlay
                    const overlay = document.createElement('DIV');
                    overlay.appendChild(imagen);
                    overlay.classList.add("overlay");

                    overlay.onclick = function() {
                        overlay.remove();
                    }

                    //-- Mostrar galeria
                    const body = document.querySelector('BODY');
                    body.appendChild(overlay);

                    //-- Bot√≥n cerrar
                    const botonCerrar = document.createElement("P");
                    botonCerrar.textContent = "‚úñ";
                    botonCerrar.classList.add("boton-cerrar");

                    overlay.appendChild(botonCerrar);

                    botonCerrar.onclick = cerrar;

                    function cerrar() {
                        overlay.remove();
                    }
                }
