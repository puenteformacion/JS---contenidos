3.3. EXPRESIONES

              
                /*
                 *  Tema: Expresiones
                 *  Autor   : JL Puente
                 *  Versión : EcmaScript ES11
                 */

                // -------------------------
                // Operadores de Asignación
                // -------------------------

                console.warn('--- Operadores de Asignación ---');

                var numero = 10;
                console.log(`Numero = ${numero}`);

                numero = 25; // Reasignar
                console.log(`Numero = ${numero}`);

                numero = numero + 15; // Reasignar con un incremento aditivo
                console.log(`Numero = ${numero}`);

                numero += 5; // Reasignar con un incremento aditivo (otra sintaxis)
                console.log(`Numero = ${numero}`);

                numero -= 20; // Reasignar con un decremento
                console.log(`Numero = ${numero}`);

                numero *= 4; // Reasignar con un incremento multiplicativo
                console.log(`Numero = ${numero}`);

                numero /= 2; // Reasignar con una partición
                console.log(`Numero = ${numero}`);

                numero %= 6; // Reasignar con un módulo
                console.log(`Numero = ${numero}`);

                // ---------------------
                // Pre y Post Incremento
                // ---------------------

                console.warn('--- Operadores de Pre y Post Incremento ---');

                var i = 10;

                console.log(`Pre Incremento: i = ${++i}`); // Pre Incremento: Se incrementa en la misma instrucción

                i = 10;
                console.log(`Post Incremento: i = ${i++}`); // Post Incremento: se incrementa en la siguiente instrucción
                console.log(`Post Incremento: i = ${i}`);

                // ----------------------
                // Operadores Aritméticos
                // ----------------------

                console.warn('--- Operaciones Aritméticas ---');

                var numero1 = 20, numero2 = 8;

                var suma = numero1 + numero2; // Sumar
                console.log(`suma = ${suma}`);

                var resta = numero1 - numero2; // Restar
                console.log(`resta = ${resta}`);

                var producto = numero1 * numero2; // Multiplicar
                console.log(`producto = ${producto}`);

                var cociente = numero1 / numero2; // Dividir
                console.log(`cociente = ${cociente}`);

                var cocienteEntero = numero1 / numero2 >> 0; // Dividir con resultado entero
                console.log(`cocienteEntero = ${cocienteEntero}`);

                var modulo = numero1 % numero2; // Modulo de la división entera
                console.log(`modulo = ${modulo}`);

                // -----------------------
                // Operadores Relacionales
                // -----------------------

                console.warn('--- Operadores Relacionales ---');

                var numero3 = 30;
                var cadena = "20";

                console.log(`${numero1} > ${numero2} : ${numero1 > numero2}`);
                console.log(`${numero1} >= ${numero2} : ${numero1 >= numero2}`);
                console.log(`${numero1} < ${numero2} : ${numero1 < numero2}`);
                console.log(`${numero1} == ${cadena} : ${numero1 == cadena}`);
                console.log(`${numero1} === ${numero2} : ${numero1 == numero2}`);

                // ------------------
                // Operadores Lógicos
                // ------------------

                console.warn('--- Operadores Lógicos ---');

                console.log(`(${numero1} > ${numero2}) && (${numero1} < ${numero3}) : ${(numero1 > numero2) && (numero1 < numero3)}`);
                console.log(`(${numero1} < ${numero2}) || (${numero1} != ${cadena}) : ${(numero1 < numero2) || (numero1 != cadena)}`);
                console.log(`![(${numero1} < ${numero2}) || (${numero1} == ${numero3})] : ${!(numero1 < numero2) || (numero1 == numero3)}`);

                // ------------------------
                // Operadores Condicionales
                // ------------------------

                console.warn('--- Operadores Condicionales ---');

                console.log(`${numero1} >= ${numero2} : ${numero1 >= numero2 ? "OK":"MAL"}`);

3.5. CONVERSIÓN DE TIPOS

              
                /*
                 *  Tema: Conversión de Tipos
                 *  Autor   : JL Puente
                 *  Version : EcmaScript ES11
                 */

                //-- Casting o Conversión de Tipos en la entrada

                var cadena = prompt('Ingrese una cadena: ');
                var numero1 = parseInt(prompt('Ingrese un número entero: '));
                var numero2 = parseFloat(prompt('Ingrese un número decimal: '));

                console.log(`Cadena : ${cadena}`);
                console.log(`Numero1 = ${numero1}`);
                console.log(`Numero2 : ${numero2}`);

                var numero3 = parseInt(prompt("Ingrese un número entero por segunda vez: "));
                var numero4 = parseInt(prompt("Ingrese un número entero por tercera vez: "));

                console.log(`Numero3 = ${numero3}`);
                console.log(`Numero4 : ${numero4}`);

                //-- Casting o Conversión de Tipos en la salida 

                console.warn('--- Casting ---');

                numero3 = numero3 + ""; // convertir número a cadena forma 1
                numero3 = String(numero3); // convertir número a cadena forma 2
                numero4 = parseFloat(numero4); // convertir número entero a punto flotante

                console.log(`Número => Cadena : ${numero3}`);
                console.log(`Número => Número : ${numero4.toFixed(3)}`);
