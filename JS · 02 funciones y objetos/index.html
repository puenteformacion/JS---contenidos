Métodos de una Cadena

              
                console.warn('-- charAt --');
                console.log(`dinosaurio(0) : ${animal.charAt(0)}`); //--> Extraer caracter
                console.log(`dinosaurio(3) : ${animal.charAt(3)}`);
                console.log(`dinosaurio(5) : ${animal.charAt(5)}`);
                console.log(`dinosaurio(7) : ${animal.charAt(7)}`);

                console.warn('-- concat --');
                console.log(`${animal.concat(" rosa")}`); //--> Otra forma de concatenar


                console.warn('-- indexOf --');
                console.log(`dinosaurio | dino : ${animal.indexOf("dino")}`); //--> Busqueda en la cadena, devuelve la posición
                console.log(`dinosaurio | sa : ${animal.indexOf("sa")}`);
                console.log(`dinosaurio | rio : ${animal.indexOf("rio")}`);


                console.warn('-- repeat --');
                console.log(`Repetir(x3) : ${animal.repeat(3)}`); //--> Repetición de la cadena
                console.log(`Repetir(x5) : ${animal.repeat(5)}`);
                console.log(`Repetir(x0) : ${animal.repeat(0)}`);
                        

                console.warn('-- replace --');
                console.log(`dinosaurio : ${animal.replace("dino", "DINO")}`); //--> Reemplazar en la cadena
                console.log(`dinosaurio : ${animal.replace("dino", "plateo")}`);


                console.warn('-- slice + substring --');
                console.log(`dinosaurio : ${animal.slice(3,7)}`); //--> Extraer parte de la cadena
                console.log(`dinosaurio : ${animal.slice(3,-2)}`);
                console.log(`dinosaurio : ${animal.substring(0,4)}`); //--> Extrae una subcadena
                console.log(`dinosaurio : ${animal.substring(3)}`);


                console.warn('-- split --');
                let animales = "gato, perro, loro, gallo"; 
                let arreglo = animales.split(', '); //--> Forma un arreglo a partir de una cadena
                console.log(arreglo);
                console.log(`[1] : ${arreglo[0]}`);
                console.log(`[2] : ${arreglo[1]}`);
                console.log(`[3] : ${arreglo[2]}`);
                console.log(`[4] : ${arreglo[3]}`);


                console.warn('-- startsWith --');
                console.log(`dinosaurio | sa : ${animal.startsWith("sa")}`); //--> ¿Empieza en...?
                console.log(`dinosaurio | di : ${animal.startsWith("di")}`);


                console.warn('-- text-transform --');
                console.log(`dinosaurio : ${animal.toUpperCase()}`); //--> convertir a mayúsculas / toLowerCase: a minúsculas


                console.warn('-- toString --');
                console.log(`${animal.toString()}`); //--> Escribir cadena


                console.warn('-- trim --');   
                let texto = 'Este           texto es       un poco       raro'; 
                texto = texto.trim(); //--> Elimina los espacios, sólo funciona en el DOM
                console.log(`${texto}`);
                document.write(texto.trim() + "<br>");

Métodos de un array

              
                console.warn('--- Métodos del arreglo ---');

                //---- Crear Array a partir de un objeto
                console.log(`'BARCELONA' => ${Array.from('BARCELONA')}`);

                //---- Comprobar si es Array
                console.log(`¿Es array? : ${Array.isArray(frutas)}`);

                //---- Crear una nueva instancia de Array con variables pasadas como elementos
                console.log(`(1,2,3) => ${Array.of(1, 2, 3)}`);

                //---- Añadir elemento al final de un Array
                console.log(`numeros = [10, 20, 30, 40, 50]`);
                numeros.push(60);
                console.log(numeros);

                //---- Eliminar elemento al final de un Array 
                numeros.pop();
                console.log(numeros);

                //---- Añadir elemento del inicio de un Array 
                numeros.unshift(0);
                console.log(numeros);

                //---- Eliminar elemento al inicio de un Array 
                numeros.shift();
                console.log(numeros);

                //---- Inviertir el orden de los elementos 
                console.log(numeros.reverse());

                //---- Ordenar los elementos de un Array
                console.log(frutas.sort());

                //---- Encontrar el índice de un elemento en el Array 
                console.log(vocales.indexOf("e"));
                console.log(vocales.indexOf("u"));

                //---- Eliminar un único elemento con el índice de posición 
                console.log(frutas.splice(2,1));
                console.log(frutas.splice(2,2));

                //---- Concatenar Arrays 
                console.log(vocales.concat(numeros));

                //---- Unir los elementos en una cadena
                console.log(numeros.join());

                //---- ToString()
                console.log(numeros.toString());
                console.log(frutas.toString());

                //---- Escribir uno a uno los elementos
                numeros.forEach(elemento => console.log(elemento));

                //---- Filtrar
                console.log(frutas.filter(elemento => elemento.length > 5));

5. OBJETOS

Creando un objeto

              
                const Persona = {
                    nombre: 'Marco',
                    apellidos: 'Martinez Gómez',
                    profesion: 'Diseñador Gráfico',
                    email: 'marmaro@gmail.com',
                    edad: 26,
                    musica: ['rock','clásica','grunge', 'metal'],
                    hogar: {
                        ciudad: 'Valencia',
                        pais: 'España'
                    },
                    born: function() {
                        return new Date().getFullYear() - this.edad;
                    }
                }

Instanciando un objeto

              
                console.log(`Nombre : ${Persona.nombre} - Apellido : ${Persona.apellidos}`);
                console.log(`Profesion : ${Persona.profesion}`);
                console.log(`Email : ${Persona.email}`);
                console.log(`Edad : ${Persona.edad}`);
                console.log(`Música : ${Persona.musica}`);
                console.log(`Ciudad : ${Persona.hogar.ciudad} - País: ${Persona.hogar.pais}`);
                console.log(`Año de Nacimiento : ${Persona.born()}`);
